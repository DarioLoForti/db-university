QUERY CON GROUP BY

1. Contare quanti iscritti ci sono stati ogni anno

SELECT COUNT(id) AS totale_iscritti,YEAR(enrolment_date) AS anno 
FROM students 
GROUP BY YEAR(enrolment_date);
